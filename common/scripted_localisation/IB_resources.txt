defined_text = {
	name = IB_OIL_TEXT
	text = {
		trigger = {
			check_variable = {resource_imported@oil > 0 }
			check_variable = {resource@oil > resource_imported@oil }
		}
		localization_key = oil_yellow_tt
	}
	text = {
		trigger = {
			always=yes
		}
		localization_key = oil_green_tt
	}
}
